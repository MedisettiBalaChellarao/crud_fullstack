pipeline {
    agent any

    environment {
        // Tomcat deployment details
        TOMCAT_URL = 'http://13.223.123.83:9090/manager/text'
        TOMCAT_USER = 'admin'
        TOMCAT_PASS = 'admin'

        // Paths
        BACKEND_DIR = 'crud_back'
        FRONTEND_DIR = 'crud_front'

        // WAR files
        BACKEND_WAR = 'crud_back/target/springapp1.war'
        FRONTEND_WAR = 'crud_front/frontapp1.war'
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/MedisettiBalaChellarao/crud_fullstack.git'
            }
        }

        stage('Build Backend') {
            steps {
                dir("${BACKEND_DIR}") {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Deploy Backend') {
            steps {
                script {
                    sh """
                    curl -u $TOMCAT_USER:$TOMCAT_PASS --upload-file $BACKEND_WAR \
                    "$TOMCAT_URL/deploy?path=/springapp1&update=true"
                    """
                }
            }
        }

        stage('Build Frontend') {
            steps {
                dir("${FRONTEND_DIR}") {
                    sh 'npm install'
                    sh 'npm run build'
                    
                    // Package React build folder into WAR
                    sh 'jar -cvf frontapp1.war -C dist .'
                }
            }
        }

        stage('Deploy Frontend') {
            steps {
                script {
                    sh """
                    curl -u $TOMCAT_USER:$TOMCAT_PASS --upload-file $FRONTEND_WAR \
                    "$TOMCAT_URL/deploy?path=/frontapp1&update=true"
                    """
                }
            }
        }
    }
}
